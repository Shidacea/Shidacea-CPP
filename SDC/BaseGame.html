def initialize
	# Distance unit in pixels
	@meter = 50.0

	# Time unit in frames
	@second = 60.0

	# Physics time step
	@dt = 1.0

	# Gravity
	@gravity = SDC::Coordinates.new(0, 30.0) * (@meter / @second**2)

	# Basic hash for global and local switches
	@switches = {}

	# Basic hash for global and local variables
	@variables = {}
end
